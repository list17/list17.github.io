2、 这道题我采用的较为复杂，在这里写一个简单思路，利用递归函数在constructor里面构造出这个数组，之后的next_value()以及has_next()只需要读取该数组便可，利用递归函数先序遍历一个树较为简单。也可以使用栈结构进行遍历。自己的方法过于麻烦了，也是使用栈结构。在这里同样附上大佬的generator方法： 


    'use strict';

    class TreeIterator {
        constructor(tree) {
            function* gen(node) {
                yield node.val
                for (let child of node.children)
                    yield* gen(child);
            }
            this.iter = gen(tree);
            this.curr = this.iter.next();
        }
        has_next() {
            return !this.curr.done;
        }
        next_value() {
            const value = this.curr.value;
            this.curr = this.iter.next();
            return value;
        }
    }
